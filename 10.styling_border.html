border in Action

  • /* 10px of border applied in all directions */
    border: 10px;
  • /* 10px of border applied vertically and 0px applied horizontally */
    border: 10px 0;
  • /* 8px of border applied to the top */
    border-top: 8px;
  • /* 8px of border applied to the right */
    border-right: 8px;
  • /* 8px of border applied to the bottom */
    border-bottom: 8px;
  • /* 8px of border applied to the left */
    border-left: 8px;
  • /* border-radius applied in all radius */
    border-radius: 8px;
  • /* border style applied */
    border: 8px solid red; 
li {
    border: 8px;
    }
